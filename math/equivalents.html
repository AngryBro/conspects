\( \Huge = \)

  1. \( \begin{array}[t]{|l|} \hline \displaystyle a_n t^n + a_{n-1} t^{n-1} + \ldots + a_2 t^2 + a_1 t + a_0 = 0 \Leftrightarrow \\ \Leftrightarrow (t - t_1) \cdot(t - t_2) \cdot \ldots \cdot (t - t_m) = 0 \Leftrightarrow \left[ \begin{array}{l} t=t_1 \\ t=t_2 \\ \ldots \\ t=t_m \end{array} \right. \\ \hline \end{array} \)
    • \( a t^2 + b t + c = 0 \Leftrightarrow \left[ \begin{array}{l} t = t_1 \\ t = t_2 \end{array} \right. \)
  2. \( \begin{array}[t]{|l|} \hline \displaystyle a_1 \cdot a_2 \cdot \ldots \cdot a_n = 0 \Leftrightarrow \left\{ \begin{array}{l} \left[ \begin{array}{l} a_1 = 0 \\ a_2 = 0 \\ \ldots \\ a_n = 0 \end{array} \right. \\ ограничения~из~a_1,~a_2, \ldots ,a_n \end{array} \right. \\ \hline \end{array} \)
    • \( a \cdot b = 0 \Leftrightarrow \left[ \begin{array}{l} \left\{ \begin{array}{l} a = 0 \\ ограничения~из~b \end{array} \right. \\ \left\{ \begin{array}{l} b = 0 \\ ограничения~из~a \end{array} \right. \end{array} \right. \)
  3. \(\begin{array}[t]{|l|} \hline \displaystyle \frac{a}{b} = \frac{c}{d} \Leftrightarrow \left\{ \begin{array}{l} a \cdot d = b \cdot c \\ b \neq 0 \\ d \neq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \displaystyle \frac{a}{b} = 0 \Leftrightarrow \left\{ \begin{array}{l} a = 0 \\ b \neq 0 \end{array} \right. \)
  4. \(\begin{array}[t]{|l|} \hline \displaystyle \operatorname{МонотоннаяФункция}{(a)} = \operatorname{МонотоннаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a=b \\ ограничение~на~a~или~b \end{array} \right. \\ \hline \end{array} \)
    • \( c^a = c^b ~~\overset{c = const\neq 1}{\Longleftrightarrow}~~ a = b \)
    • \( \log_c{a} = \log_c{b} ~~\overset{c = const}{\Longleftrightarrow}~~ \left\{ \begin{array}{l} a = b \\ a > 0 \end{array} \right. \)
    • \( \sqrt[2n]{a} = \sqrt[2n]{b} \Leftrightarrow \left\{ \begin{array}{l} a = b \\ a \geq 0 \end{array} \right. \)
    • \( \sqrt[2n+1]{a} = \sqrt[2n+1]{b} \Leftrightarrow a = b \)
    • \( a^{2n+1} = b^{2n+1} \Leftrightarrow a = b \)
  5. \(\begin{array}[t]{|l|} \hline \displaystyle \sqrt[2n]{a} = b \Leftrightarrow \left\{ \begin{array}{l} a = b^{2n} \\ b \geq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \sqrt{a} = b \Leftrightarrow \left\{ \begin{array}{l} a = b^2 \\ b \geq 0 \end{array} \right. \)
  6. \( \begin{array}[t]{|l|} \hline a_n x^n + a_{n-1} x^{n-1}y + a_{n-2}x^{n-2}y^2 + \ldots +a_1xy^{n-1}+a_0y^n=0 \\ \hline \end{array} \)
    • \( a \cdot m^{t} = b \cdot n^{t} ~\Leftrightarrow ~ a \cdot \left(\displaystyle\frac{m}{n}\right)^t = b \)
    • \( a \cdot \left(m^2\right)^{t} + b \cdot (mn)^{t} + c \cdot \left(n^2\right)^{t} = 0 ~\Leftrightarrow ~ a \cdot \left(\left(\displaystyle\frac{m}{n}\right)^t\right)^2 + b \cdot \left(\displaystyle\frac{m}{n}\right)^t + c = 0 \)
    • \( a \cdot \sin{t} = b \cdot \cos{t} ~\Leftrightarrow~ a \cdot \operatorname{tg}{t} = b \)
    • \( a \cdot \sin^2{t} + b \cdot \sin{t} \cdot \cos{t} + c \cdot \cos^2{t} = 0 ~\Leftrightarrow ~ a \cdot \operatorname{tg^2}{t} + b \cdot \operatorname{tg}{t} + c = 0 \)

\( \Huge \neq \)

  1. \( \begin{array}[t]{|l|} \hline a_1 \cdot a_2 \cdot \ldots \cdot a_n \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a_1 \neq 0 \\ a_2 \neq 0 \\ \ldots \\ a_n \neq 0 \end{array} \right. \\ \hline \end{array} \)
  2. \( \begin{array}[t]{|l|} \hline \displaystyle \frac{a}{b} \neq \frac{c}{d} \Leftrightarrow \left\{ \begin{array}{l} a \cdot d \neq b \cdot c \\ b \neq 0 \\ d \neq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \displaystyle \frac{a}{b} \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a \neq 0 \\ b \neq 0 \end{array} \right. \)
  3. \( \begin{array}[t]{|l|} \hline \operatorname{МонотоннаяФункция}{(a)} \neq \operatorname{МонотоннаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ ограничение~на~a \\ ограничение~на~b \end{array} \right. \\ \hline \end{array} \)
    • \( c^a \neq c^b ~~\overset{c = const\neq 1}{\Longleftrightarrow}~~ a \neq b \)
    • \( \log_c{a} \neq \log_c{b} ~~\overset{c = const}{\Longleftrightarrow}~~ \left\{ \begin{array}{l} a \neq b \\ a > 0 \\ b > 0 \end{array} \right. \)
    • \( \sqrt[2n]{a} \neq \sqrt[2n]{b} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ a \geq 0 \\ b \geq 0 \end{array} \right. \)
    • \( \sqrt[2n+1]{a} \neq \sqrt[2n+1]{b} \Leftrightarrow a \neq b \)
    • \( a^{2n+1} \neq b^{2n+1} \Leftrightarrow a \neq b \)
  4. \(\begin{array}[t]{|l|} \hline \sqrt[2n]{a} \neq b \Leftrightarrow \left[ \begin{array}{l} \begin{cases} b < 0 \\ a \geq 0 \end{cases} \\ \begin{cases} b \geq 0 \\ a \geq 0 \\ a \neq b^{2n} \end{cases} \end{array} \right. \\ \hline \end{array} \)

\( \Huge > \normalsize,~ \Huge< \normalsize,~ \Huge \geq \normalsize,~ \Huge \leq \)

  1. \(\begin{array}[t]{|l|} \hline \displaystyle \frac{(x-a_1)^{k_1} \cdot (x-a_2)^{k_2} \cdot \ldots \cdot (x-a_n)^{k_n}} {(x-b_1)^{t_1} \cdot (x-b_2)^{t_2} \cdot \ldots \cdot (x-b_m)^{t_m}} \vee 0 - по~методу~интервалов \\ \hline \end{array} \)
    • \( \displaystyle \frac{1}{(x - a_1)^{k_1}\cdot(x - a_2)^{k_2}\cdot\ldots\cdot(x - a_n)^{k_n}} \vee 0 \)
    • \( \displaystyle (x - a_1)^{k_1}\cdot(x - a_2)^{k_2}\cdot\ldots\cdot(x - a_n)^{k_n} \vee 0 \)
  2. \( \begin{array}[t]{|l|} \hline \operatorname{ВозрастающаяФункция}{(a)} \vee \operatorname{ВозрастающаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \vee b \\ ограничение~на~a \\ ограничение~на~b \end{array} \right. \\ \hline \end{array} \)
    • \( c^a \leq c^b ~~\overset{c=const>1}{\Longleftrightarrow}~~ a \leq b \)
    • \( \log_{c}{a} \leq \log_{c}{b} ~~\overset{c = const>1}{\Longleftrightarrow}~~ \left\{ \begin{array}{l} a \leq b \\ a > 0 \end{array} \right. \)
    • \( \sqrt[2n]{a} \leq \sqrt[2n]{b} \Leftrightarrow \left\{ \begin{array}{l} a \leq b \\ a \geq 0 \end{array} \right. \)
    • \( \sqrt[2n+1]{a} \leq \sqrt[2n+1]{b} \Leftrightarrow a \leq b \)
    • \( a^{2n+1} \leq b^{2n+1} \Leftrightarrow a \leq b \)
  3. \( \begin{array}[t]{|l|} \hline \operatorname{УбывающаяФункция}{(a)} \vee \operatorname{УбывающаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \land b \\ ограничение~на~a \\ ограничение~на~b \end{array} \right. \\ \hline \end{array} \)
    • \( c^a \leq c^b ~~\overset{c=const < 1 }{\Longleftrightarrow}~~ a \geq b \)
    • \( \log_{c}{a} \leq \log_{c}{b} ~~\overset{c = const<1}{\Longleftrightarrow}~~ \left\{ \begin{array}{l} a \geq b \\ b > 0 \end{array} \right. \)
  4. Метод рационализации
    1. Справа 0
    2. Слева произведение
    3. Выбранные множители представлены в виде разности одинаковой монотонной функции
    В общем виде:
    • \( \frame{ \left(ВозрастающаяФункция(a)- ВозрастающаяФункция(b) \right)^{\pm n} \cdot X \vee 0 \Leftrightarrow \cases { (a-b)^{\pm n} \cdot X \vee 0 \\ ограничение~на~a \\ ограничение~на~b } } \)
    • \( \frame{ \left(УбывающаяФункция(a)- УбывающаяФункция(b) \right)^{\pm n} \cdot X \vee 0 \Leftrightarrow \cases { (b-a)^{\pm n} \cdot X \vee 0 \\ ограничение~на~a \\ ограничение~на~b } } \)
    Пример:
    \( \displaystyle \frac{(\log_{2}{a} - \log_{2}{b})\cdot(0.9^c - 0.9^d)} {(\sqrt{x} - \sqrt{y})\cdot(z^{2n+1} - t^{2n+1})} \vee 0 \Leftrightarrow \cases{ \displaystyle \frac{(a-b) \cdot (d - c)}{(x-y) \cdot (z-t)} \vee 0 \\ a > 0 \\ b > 0 \\ x \geq 0 \\ y \geq 0 } \)
  5. \( \begin{array}[t]{|l|} \hline \sqrt[2n]{a} \vee b \overset{\vee \in \{>,\geq\}}{\Longleftrightarrow} \left[ \begin{array}{l} \begin{cases} b < 0 \\ a \geq 0 \end{cases} \\ \begin{cases} b \geq 0 \\ a \vee b^{2n} \end{cases} \end{array} \right. \\ \hline \end{array} \)
  6. \( \begin{array}[t]{|l|} \hline \sqrt[2n]{a} \vee b \overset{\vee \in \{ < ,\leq\}}{\Longleftrightarrow} \begin{cases} b \geq 0 \\ a \geq 0 \\ a \vee b^{2n} \end{cases} \\ \hline \end{array} \)

\( \Huge \{ \normalsize,~ \Huge [ \)

  2. \(\begin{array}[t]{|l|} \hline \left[ \begin{array}{l} A \\ A \end{array} \right. \Leftrightarrow A \\ \hline \end{array} \)
  3. \(\begin{array}[t]{|l|} \hline \begin{cases} \begin{cases} A \\ B \end{cases} \\ C \end{cases} \Leftrightarrow \begin{cases} A \\ B \\ C \end{cases}\\ \hline \end{array} \)
  4. \(\begin{array}[t]{|l|} \hline \left[ \begin{array}{l} \left[ \begin{array}{l} A \\ B \end{array}\right. \\ C \end{array}\right. \Leftrightarrow \left[ \begin{array}{l} A \\ B \\ C \end{array}\right. \\ \hline \end{array} \)
  5. \(\begin{array}[t]{|l|} \hline \begin{cases} \left[ \begin{array}{l} A \\ B \end{array} \right. \\ C \end{cases} ~\Leftrightarrow~ \left[ \begin{array}{l} \begin{cases} A \\ C \end{cases} \\ \begin{cases} B \\ C \end{cases} \end{array} \right. \\ \hline \end{array} \)
  6. \(\begin{array}[t]{|l|} \hline \begin{cases} A \\ B \end{cases} ~~~\overset{A \Rightarrow B}{\Longleftrightarrow} ~~~A\\ \hline \end{array} \)
  7. \( \begin{array}[t]{|l|} \hline \displaystyle ложь \Leftrightarrow x \not \in \mathbb{R} \\ \hline \end{array} \)
  8. \( \begin{array}[t]{|l|} \hline истина \Leftrightarrow ограничения \\ \hline \end{array} \)
    • \( истина~без~ограничений \Leftrightarrow x \in \mathbb{R} \)
  9. \( \begin{array}[t]{|l|} \hline \left\{ \begin{array}{l} A \\ x \not \in \mathbb{R} \end{array} \right. \Leftrightarrow x \not \in \mathbb{R} \\ \hline \end{array} \)
